var egm = egrid.egm({
  enableClickNode: true,
  nodeColor: function(node) {
    return node.color;
  },
  nodeOpacity: function(node) {
    return node.opacity;
  },
  nodeScale: function(node) {
    return node.size;
  },
  nodeVisibility: function(node) {
    return node.visible;
  }
});
d3.select('svg.display')
  .datum(grid)
  .call(egm.css())
  .call(egm);
